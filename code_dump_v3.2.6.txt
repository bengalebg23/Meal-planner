let recipes = {};

function initRecipes() {
  try {
    const saved = localStorage.getItem('mealplanner_recipes');
    if (saved) {
      recipes = JSON.parse(saved);
      Object.keys(DEFAULT_RECIPES).forEach(name => {
        if (!recipes[name]) recipes[name] = JSON.parse(JSON.stringify(DEFAULT_RECIPES[name]));
        // Backfill tag on old saved recipes that predate tagging
        if (!recipes[name].tag) recipes[name].tag = 'ClaudeRecipe';
      });
    } else {
      recipes = JSON.parse(JSON.stringify(DEFAULT_RECIPES));
    }
  } catch(e) { recipes = JSON.parse(JSON.stringify(DEFAULT_RECIPES)); }
}

function saveRecipes() {
  try {
    localStorage.setItem('mealplanner_recipes', JSON.stringify(recipes));
    const ind = document.getElementById('save-indicator');
    if (ind) { ind.style.display = 'inline'; clearTimeout(ind._t); ind._t = setTimeout(() => ind.style.display = 'none', 1500); }
  } catch(e) {}
}

function getDay(key) {
  return DAYS.find(d => d.key === key) || customDays[key] || {key, label: key, isWeekday: false};
}

function emptyPlan() {
  const p = {};
  DAYS.forEach(d => {
    let work = "";
    if (d.key === "mon" || d.key === "wed") work = "office";
    else if (d.key === "sat" || d.key === "sun") work = "weekend";
    else if (d.isWeekday) work = "wfh";
    p[d.key] = {special:"", work, meals:{}};
    FAMILY.forEach(person => {
      let L = "", T = "";
      if (person === "Reuben" && d.isWeekday) L = "School";
      if (person === "Reuben" && (d.key === "mon" || d.key === "thu")) T = "Karen's";
      if (person === "Vivien" && d.key === "mon") { L = "Karen's"; T = "Karen's"; }
      if (person === "Vivien" && (d.key === "thu" || d.key === "fri")) { L = "Nursery"; T = "Nursery"; }
      if (person === "Ben" && d.key === "mon") L = "MIRA";
      if (person === "Emily" && (d.key === "mon" || d.key === "thu")) L = "Sandwich";
      p[d.key].meals[person] = {L, T};
    });
  });
  return p;
}

function addCustomDay() {
  const DAY_NAMES = ["sun","mon","tue","wed","thu","fri","sat"];
  const DAY_LABELS = {sun:"Sun",mon:"Mon",tue:"Tue",wed:"Wed",thu:"Thu",fri:"Fri",sat:"Sat"};
  const weekOffset = getWeekOffset();
  const now = new Date();
  const todayIdx = now.getDay();

  // Get date of first row - handles standard days and previous custom days
  let firstDate = null;
  const firstKey = activeDays[0];
  const firstStdIdx = DAY_NAMES.indexOf(firstKey);
  if (firstStdIdx !== -1) {
    const diff = firstStdIdx - todayIdx;
    firstDate = new Date(now);
    firstDate.setDate(now.getDate() + diff + (weekOffset * 7));
  } else if (customDays[firstKey] && customDays[firstKey].dateOffset !== undefined) {
    firstDate = new Date(now);
    firstDate.setDate(now.getDate() + customDays[firstKey].dateOffset + (weekOffset * 7));
  }

  const newKey = "custom_" + Date.now();
  let newLabel = "Day";
  let dateOffset = null;
  if (firstDate) {
    const prevDate = new Date(firstDate);
    prevDate.setDate(firstDate.getDate() - 1);
    const prevDayIdx = prevDate.getDay();
    newLabel = DAY_LABELS[DAY_NAMES[prevDayIdx]];
    const todayMidnight = new Date(now); todayMidnight.setHours(0,0,0,0);
    const prevMidnight = new Date(prevDate); prevMidnight.setHours(0,0,0,0);
    dateOffset = Math.round((prevMidnight - todayMidnight) / (24*60*60*1000)) - (weekOffset * 7);
  }
  customDays[newKey] = {key: newKey, label: newLabel, dateOffset: dateOffset, isWeekday: false};
  activeDays.unshift(newKey);
  ensureDayInPlan(newKey);
  saveToStorage();
  renderTable();
}

function removeDay(key) {
  activeDays = activeDays.filter(k => k !== key);
  delete customDays[key];
  saveToStorage();
  renderTable();
}

function loadFromStorage() {
  try {
    const savedPlan = localStorage.getItem('mealplanner_plan');
    if (savedPlan) {
      const parsed = JSON.parse(savedPlan);
      // Merge saved data into fresh plan (preserves defaults for any new keys)
      const fresh = emptyPlan();
      Object.keys(parsed).forEach(key => {
        if (!fresh[key]) { fresh[key] = {special:"", work:"", meals:{}}; FAMILY.forEach(p => { fresh[key].meals[p] = {L:"",T:""}; }); }
        if (parsed[key]) {
          fresh[key].special = parsed[key].special || "";
          fresh[key].work = parsed[key].work !== undefined ? parsed[key].work : fresh[key].work;
          FAMILY.forEach(p => {
            if (parsed[key].meals && parsed[key].meals[p]) {
              fresh[key].meals[p].L = parsed[key].meals[p].L !== undefined ? parsed[key].meals[p].L : fresh[key].meals[p].L;
              fresh[key].meals[p].T = parsed[key].meals[p].T !== undefined ? parsed[key].meals[p].T : fresh[key].meals[p].T;
            }
          });
        }
      });
      plan = fresh;
    }
    const savedNotes = localStorage.getItem('mealplanner_notes');
    if (savedNotes !== null) document.getElementById('notes-ta').value = savedNotes;
    const savedWeek = localStorage.getItem('mealplanner_week');
    if (savedWeek) document.getElementById('week-label').value = savedWeek;
    const savedActiveDays = localStorage.getItem('mealplanner_activedays');
    if (savedActiveDays) activeDays = JSON.parse(savedActiveDays);
    const savedCustomDays = localStorage.getItem('mealplanner_customdays');
    if (savedCustomDays) customDays = JSON.parse(savedCustomDays);
  } catch(e) { console.warn('Could not load:', e); }
}


function getWeekOffset() {
  if (!viewingWeekKey) return 0;
  const cur = getCurrentWeekKey();
  if (!cur) return 0;
  // Parse week keys to actual Monday dates and diff in weeks
  function weekKeyToMonday(key) {
    const [y, w] = key.split('_W').map(Number);
    // ISO week 1 is the week containing the first Thursday of the year
    const jan4 = new Date(y, 0, 4);
    const startOfWeek1 = new Date(jan4);
    startOfWeek1.setDate(jan4.getDate() - ((jan4.getDay() + 6) % 7));
    const d = new Date(startOfWeek1);
    d.setDate(d.getDate() + (w - 1) * 7);
    return d;
  }
  const curDate = weekKeyToMonday(cur);
  const viewDate = weekKeyToMonday(viewingWeekKey);
  return Math.round((viewDate - curDate) / (7 * 24 * 60 * 60 * 1000));
}

function getDayDate(dayKey) {
  const now = new Date();
  const dayOfWeek = now.getDay(); // 0=Sun
  const days = {mon:1,tue:2,wed:3,thu:4,fri:5,sat:6,sun:0};
  if (!(dayKey in days)) return '';
  const target = days[dayKey];
  const diff = target - dayOfWeek;
  const d = new Date(now);
  d.setDate(now.getDate() + diff + (getWeekOffset() * 7));
  return d.getDate() + '/' + (d.getMonth()+1) + '(' + getWeekOffset() + ')';
}

function renderTableHighlights() {
  document.querySelectorAll(".meal-cell").forEach(td => {
    const inp = td.querySelector("textarea");
    if (!inp) return;
    const key = inp.dataset.cellkey;
    if (multiSelected.has(key)) {
      td.classList.add("selected");
    } else {
      td.classList.remove("selected");
    }
  });
}

function updateTableHighlight() {
  document.querySelectorAll(".meal-cell").forEach(td => {
    td.classList.remove("active");

    if (inp) inp.classList.remove("active-input");

  // Re-render is simplest for highlight
  renderTable();
}

let selectedOrderIndex = null;

function scoreOrder(order, plannedKeywords) {
  const orderText = order.items.join(' ').toLowerCase();
  let matchScore = 0;
  plannedKeywords.forEach(kw => { if (orderText.includes(kw)) matchScore++; });
  const sizeBonus = order.items.length * 0.4; // weight towards bigger shops
  return matchScore * 3 + sizeBonus;
}

function getPlannedKeywords() {
  const meals = typeof getPlannedMeals === 'function' ? getPlannedMeals() : [];
  const stopWords = new Set(['with','and','the','in','on','of','a','to','for','or','from']);
  const words = new Set();
  meals.forEach(m => {
    m.toLowerCase().split(/[\s,+\/&]+/).forEach(w => {
      if (w.length > 3 && !stopWords.has(w)) words.add(w);
    });
  });
  return [...words];
}

function autoSelectBestOrder() {
  const keywords = getPlannedKeywords();
  let bestIndex = null, bestScore = -1;
  ASDA_ORDERS.forEach((order, i) => {
    const s = scoreOrder(order, keywords);
    if (s > bestScore) { bestScore = s; bestIndex = i; }
  });
  if (bestIndex !== null) {
    selectedOrderIndex = bestIndex;
    try { localStorage.setItem('mealplanner_baseline_order', bestIndex); } catch(e) {}
    loadSelectedOrderIntoShop();
  }
  renderAsdaSection();
  renderShop();
}

function renderAsdaSection() {
  const grid = document.getElementById('asda-order-grid');
  if (!grid) return;
  grid.innerHTML = '';

  const keywords = getPlannedKeywords();
  const hasMeals = keywords.length > 0;

  // Score all orders to find the top
  const scores = ASDA_ORDERS.map((o, i) => ({
    origIndex: i,
    score: scoreOrder(o, keywords),
    matchCount: hasMeals ? keywords.filter(kw => o.items.join(' ').toLowerCase().includes(kw)).length : 0
  }));
  const topScore = Math.max(...scores.map(s => s.score));

  // Display newest-first
  const byDate = scores.slice().reverse();

  byDate.forEach(({origIndex, score, matchCount}) => {
    const order = ASDA_ORDERS[origIndex];
    const i = origIndex;
    const isSelected = selectedOrderIndex === i;
    const isTop = hasMeals && score === topScore;
    const card = document.createElement('div');
    card.className = 'asda-order-card' + (isSelected ? ' selected' : '') + (isTop && !isSelected ? ' recommended' : '');

    const matchHtml = hasMeals
      ? `<div class="aoc-match${isTop ? ' top' : ''}">${isTop ? '⭐ Best match' : matchCount + ' meal matches'}</div>`
      : '';

    card.innerHTML = `
      <div class="aoc-date">${order.label}</div>
      <div class="aoc-meta">${order.items.length} items</div>
      ${matchHtml}
      <button class="aoc-btn" onclick="selectAsdaOrder(${i})">${isSelected ? '✓ Selected' : 'Use as baseline'}</button>
    `;
    grid.appendChild(card);
  });

  const notice = document.getElementById('asda-selected-notice');
  if (notice) {
    if (selectedOrderIndex !== null) {
      const o = ASDA_ORDERS[selectedOrderIndex];
      notice.style.display = 'block';
      notice.textContent = `✓ ${o.label} loaded (${o.items.length} items) — delete what you don't need`;
    } else {
      notice.style.display = 'none';
    }
  }

  // Collapsed header badge
  const badge = document.getElementById('asda-active-badge');
  if (badge) {
    if (selectedOrderIndex !== null) {
      const o = ASDA_ORDERS[selectedOrderIndex];
      badge.style.display = 'block';
      badge.innerHTML = `<strong>${o.label}</strong> · ${o.items.length} items loaded`;
    } else {
      badge.style.display = 'none';
    }
  }
}

function loadSelectedOrderIntoShop() {
  if (selectedOrderIndex === null) return;
  const order = ASDA_ORDERS[selectedOrderIndex];
  // Map order items into shop categories using BASELINE_ITEMS cat lookup, fallback to 'other'
  const catLookup = {};
  BASELINE_ITEMS.forEach(b => { catLookup[b.text.toLowerCase()] = b.cat; });

  // Category heuristics for items not in BASELINE_ITEMS
  function guessCat(text) {
    const t = text.toLowerCase();
    if (catLookup[t]) return catLookup[t];
    if (/milk|egg|cheese|cream|butter|yogurt|spread|pastry/.test(t)) return 'dairy';
    if (/chicken|beef|mince|sausage|ham|salmon|fish|prawn|pork|lamb|chorizo/.test(t)) return 'meat';
    if (/carrot|courgette|pepper|cucumber|avocado|banana|apple|tomato|onion|broccoli|leek|spinach|kale|leaf|potato|aubergine|corn|pak choi|melon|strawberr|blueberr|grape|pear|fruit/.test(t)) return 'produce';
    if (/pasta|rice|lentil|bean|passata|tinned|tuna|coconut milk|juice|stock|soup|noodle|oat/.test(t)) return 'dry';
    if (/frozen|fish finger|sausage roll|veggie finger|pizza|chips/.test(t)) return 'frozen';
    return 'other';
  }

  // Clear existing baseline items (keep any meal-generated items that are already there)
  // Actually replace entirely — user can re-generate meals on top
  shopItems = {produce:[],meat:[],dairy:[],dry:[],frozen:[],other:[]};
  checkedItems = {};

  order.items.forEach(item => {
    const cat = guessCat(item);
    shopItems[cat].push({ text: item, id: Math.random().toString(36).slice(2) });
  });
}

function loadBaselineIntoShop() {
  // Legacy: only used if no order selected
  if (selectedOrderIndex !== null) return;
  BASELINE_ITEMS.forEach(item => {
    const cat = item.cat;
    const exists = shopItems[cat].some(i => i.text === item.text);
    if (!exists) {
      shopItems[cat].push({ text: item.text, id: Math.random().toString(36).slice(2) });
    }
  });
}

let viewingWeekKey = null; // null = current live week

function getCurrentWeekKey() {
  // ISO week number based key e.g. "2026_W16"
  const now = new Date();
  const jan4 = new Date(now.getFullYear(), 0, 4);
  const weekNum = Math.ceil(((now - jan4) / 86400000 + jan4.getDay() + 1) / 7);
  return `${now.getFullYear()}_W${String(weekNum).padStart(2,'0')}`;
}

function getArchiveKeys() {
  const keys = [];
  for (let i = 0; i < localStorage.length; i++) {
    const k = localStorage.key(i);
    if (k && k.startsWith('mealplanner_archive_')) keys.push(k.replace('mealplanner_archive_',''));
  }
  return keys.sort();
}

function saveCurrentToArchive() {
  const weekKey = getCurrentWeekKey();
  const snapshot = {
    plan: localStorage.getItem('mealplanner_plan'),
    notes: localStorage.getItem('mealplanner_notes'),
    week: localStorage.getItem('mealplanner_week'),
    activedays: localStorage.getItem('mealplanner_activedays'),
    customdays: localStorage.getItem('mealplanner_customdays'),
    shop: JSON.stringify(shopItems),
    checked: JSON.stringify(checkedItems),
  };
  try { localStorage.setItem('mealplanner_archive_' + weekKey, JSON.stringify(snapshot)); } catch(e) {}
}

      viewingWeekKey = archiveKeys[archiveKeys.length - 1];
      loadArchivedWeek(viewingWeekKey);
      if (window.fbSubscribeWeek) window.fbSubscribeWeek(viewingWeekKey);
    } else if (dir === 1) {
      viewingWeekKey = getOffsetWeekKey(1);
      loadArchivedWeek(viewingWeekKey);
      if (window.fbSubscribeWeek) window.fbSubscribeWeek(viewingWeekKey);
    }
  } else {
    const yr = parseInt(viewingWeekKey.split('_W')[0]);
    const w = parseInt(viewingWeekKey.split('_W')[1]) + dir;
    let newKey;
    if (w < 1) newKey = (yr-1) + '_W52';
    else if (w > 52) newKey = (yr+1) + '_W01';
    else newKey = yr + '_W' + String(w).padStart(2,'0');
    if (dir === 1 && newKey === currentKey) { viewingWeekKey = null; loadLiveWeek(); }
    else { viewingWeekKey = newKey; loadArchivedWeek(viewingWeekKey); if (window.fbSubscribeWeek) window.fbSubscribeWeek(newKey); }
  }
  updateWeekNavButtons();
  updateWeekNavUI();
}

function loadArchivedWeek(weekKey) {
  try {
    const raw = localStorage.getItem('mealplanner_archive_' + weekKey);
    if (!raw) {
      plan = emptyPlan();
      renderTable();
      renderBanks();
      renderShop();
      updateMealsDetected();
      return;
    }
    const snapshot = JSON.parse(raw);
    // Load into app state
    if (snapshot.plan) {
      const parsed = JSON.parse(snapshot.plan);
      const fresh = emptyPlan();
      Object.keys(parsed).forEach(key => {
        if (!fresh[key]) { fresh[key] = {special:"", work:"", meals:{}}; FAMILY.forEach(p => { fresh[key].meals[p] = {L:"",T:""}; }); }
        if (parsed[key]) {
          fresh[key].special = parsed[key].special || "";
          fresh[key].work = parsed[key].work !== undefined ? parsed[key].work : fresh[key].work;
          FAMILY.forEach(p => {
            if (parsed[key].meals && parsed[key].meals[p]) {
              fresh[key].meals[p].L = parsed[key].meals[p].L || "";
              fresh[key].meals[p].T = parsed[key].meals[p].T || "";
            }
          });
        }
      });
      plan = fresh;
    }
    if (snapshot.notes !== null) document.getElementById('notes-ta').value = snapshot.notes || '';
    if (snapshot.week) document.getElementById('week-label').value = snapshot.week;
    if (snapshot.activedays) activeDays = JSON.parse(snapshot.activedays);
    if (snapshot.customdays) customDays = JSON.parse(snapshot.customdays);
    if (snapshot.shop) shopItems = JSON.parse(snapshot.shop);
    if (snapshot.checked) checkedItems = JSON.parse(snapshot.checked);
  } catch(e) { console.warn('Could not load archived week:', e); }
  renderTable();
  renderBanks();
  renderShop();
  updateMealsDetected();
}

function loadLiveWeek() {
  plan = emptyPlan();
  loadFromStorage();
  if (selectedOrderIndex === null) loadBaselineIntoShop();
  else loadSelectedOrderIntoShop();
  renderTable();
  renderBanks();
  renderShop();
  updateMealsDetected();
  const weekKey = localStorage.getItem('mealplanner_week');
  if (weekKey && window.fbSubscribeWeek) window.fbSubscribeWeek(weekKey);
}

function updateWeekNavUI() {
  const archiveKeys = getArchiveKeys();
  const isLive = viewingWeekKey === null;
  const idx = isLive ? -1 : archiveKeys.indexOf(viewingWeekKey);

  document.getElementById('week-prev-btn').disabled = isLive ? archiveKeys.length === 0 : idx === 0;
  document.getElementById('week-next-btn').disabled = false;
  if (typeof updateWeekNavButtons === 'function') updateWeekNavButtons();
  document.getElementById('week-readonly-badge').style.display = isLive ? 'none' : 'inline';
  document.getElementById('restore-week-btn').style.display = isLive ? 'none' : 'inline-block';
  document.getElementById('week-label').readOnly = !isLive;

  // Make table inputs read-only when viewing past week
  document.querySelectorAll('.meal-input, .work-select, .flag-select, .day-label-input').forEach(el => {
    el.disabled = !isLive;
  });
}

function restoreWeek() {
  if (!viewingWeekKey) return;
  if (!confirm('Load this past week as your current plan? Your current week will be saved to the archive first.')) return;
  saveCurrentToArchive();
  // Copy archived week into live storage
  try {
    const raw = localStorage.getItem('mealplanner_archive_' + viewingWeekKey);
    if (raw) {
      const snapshot = JSON.parse(raw);
      if (snapshot.plan) localStorage.setItem('mealplanner_plan', snapshot.plan);
      if (snapshot.notes !== null) localStorage.setItem('mealplanner_notes', snapshot.notes);
      if (snapshot.week) localStorage.setItem('mealplanner_week', snapshot.week);
      if (snapshot.activedays) localStorage.setItem('mealplanner_activedays', snapshot.activedays);
      if (snapshot.customdays) localStorage.setItem('mealplanner_customdays', snapshot.customdays);
    }
  } catch(e) {}
  viewingWeekKey = null;
  loadLiveWeek();
  updateWeekNavUI();
}

// Auto-archive current week when app loads (captures previous week's data)
function autoArchiveOnLoad() {
  const weekKey = getCurrentWeekKey();
  const existing = localStorage.getItem('mealplanner_archive_' + weekKey);
  if (!existing) saveCurrentToArchive();
}

  document.getElementById('import-error').style.display = 'none';
  document.getElementById('import-ta').value = '';

  document.getElementById('share-modal').classList.remove('visible');
}

function exportWeek() {
  // Bundle the current visible week's data
  const isLive = viewingWeekKey === null;
  let snapshot;
  if (isLive) {
    snapshot = {
      week: localStorage.getItem('mealplanner_week') || '',
      plan: localStorage.getItem('mealplanner_plan') || '{}',
      notes: localStorage.getItem('mealplanner_notes') || '',
      activedays: localStorage.getItem('mealplanner_activedays') || '[]',
      customdays: localStorage.getItem('mealplanner_customdays') || '{}',
      shop: JSON.stringify(shopItems),
      checked: JSON.stringify(checkedItems),
    };
  } else {
    try {
      snapshot = JSON.parse(localStorage.getItem('mealplanner_archive_' + viewingWeekKey) || '{}');
    } catch(e) { snapshot = {}; }
  }
  const payload = 'MEALPLAN:' + btoa(JSON.stringify(snapshot));
  navigator.clipboard.writeText(payload).then(() => {
    document.getElementById('export-confirm').style.display = 'block';
  }).catch(() => {
    // Fallback for browsers that block clipboard
    document.getElementById('import-ta').value = payload;
    document.getElementById('export-confirm').textContent = 'Clipboard blocked — copy the text above manually.';
    document.getElementById('export-confirm').style.display = 'block';
  });
}

function importWeek() {
  const raw = document.getElementById('import-ta').value.trim();
  document.getElementById('import-error').style.display = 'none';
  try {
    if (!raw.startsWith('MEALPLAN:')) throw new Error('bad format');
    const snapshot = JSON.parse(atob(raw.replace('MEALPLAN:','')));
    if (snapshot.plan) localStorage.setItem('mealplanner_plan', snapshot.plan);
    if (snapshot.notes !== undefined) localStorage.setItem('mealplanner_notes', snapshot.notes);
    if (snapshot.week) localStorage.setItem('mealplanner_week', snapshot.week);
    if (snapshot.activedays) localStorage.setItem('mealplanner_activedays', snapshot.activedays);
    if (snapshot.customdays) localStorage.setItem('mealplanner_customdays', snapshot.customdays);
    viewingWeekKey = null;
    loadLiveWeek();
    updateWeekNavUI();
    document.getElementById('import-error').style.display = 'none';
    closeShareModal();
    return;
  } catch(e) {
    document.getElementById('import-error').style.display = 'block';
  }
}

function toggleLock() {
  const locked = !isPlanLocked();
  try { localStorage.setItem('mealplanner_locked', locked ? '1' : '0'); } catch(e) {}
  const btn = document.getElementById('lock-btn');
  if (btn) { btn.textContent = locked ? '🔒' : '🔓'; btn.classList.toggle('locked', locked); }
  document.querySelectorAll('.meal-input, .work-select, .flag-select, .day-label-input').forEach(el => {
    el.disabled = locked;
  });
}

function getOffsetWeekKey(n) {
  const now = new Date();
  now.setDate(now.getDate() + n * 7);
  const jan4 = new Date(now.getFullYear(), 0, 4);
  const w = Math.ceil(((now - jan4) / 86400000 + jan4.getDay() + 1) / 7);
  return now.getFullYear() + '_W' + String(w).padStart(2, '0');
}

function updateWeekNavButtons() {
  const currentKey = getCurrentWeekKey();
  const activeKey = viewingWeekKey || currentKey;
  function wkOffset(base, n) {
    const yr = parseInt(base.split('_W')[0]);
    const w = parseInt(base.split('_W')[1]) + n;
    if (w < 1) return (yr-1) + '_W52';
    if (w > 52) return (yr+1) + '_W01';
    return yr + '_W' + String(w).padStart(2,'0');
  }
  function wkLabel(k) { return 'W' + parseInt(k.split('_W')[1]); }
  const prevKey = wkOffset(activeKey, -1);
  const nextKey = wkOffset(activeKey, 1);
  const pb = document.getElementById('week-prev-btn');
  const cb = document.getElementById('week-center-btn');
  const nb = document.getElementById('week-next-btn');
  if (pb) { pb.textContent = wkLabel(prevKey); pb.className = 'week-nav-btn ' + (prevKey < currentKey ? 'past-week' : prevKey > currentKey ? 'future-week' : 'current-week'); }
  if (cb) { cb.textContent = wkLabel(activeKey); cb.className = 'week-nav-btn current-week'; }
  if (nb) { nb.textContent = wkLabel(nextKey); nb.className = 'week-nav-btn ' + (nextKey < currentKey ? 'past-week' : nextKey > currentKey ? 'future-week' : 'current-week'); }
  const restoreBtn = document.getElementById('restore-week-btn');
  if (restoreBtn) restoreBtn.style.display = viewingWeekKey ? 'inline-block' : 'none';
  const locked = isPlanLocked();
  const lockBtn = document.getElementById('lock-btn');
  if (lockBtn) { lockBtn.textContent = locked ? '🔒' : '🔓'; lockBtn.classList.toggle('locked', locked); }
}
